#Configure a Node-Profile in Routed L3-network under specific tenant

#Note 1: Replace "<tenant-name>" with the tenant name in the POST URL
#Note 2: Replace "<ext-net>" with the Routed-L3-network name in th POST URL

POST URL: http://SWITCH_IP/api/mo/uni/tn-<tenant-name>/out-<ext_net>.json
Content-Type: application/json
Cache-Control: no-cache
POST BODY :
{
	"l3extLNodeP": {
		"attributes": {
			"name": "node_Prof"
		},
		"children": [
			{
		    	"l3extRsNodeL3OutAtt": {
					"attributes": {
						"rtrId": "1.0.0.2",
						"tDn": "topology/pod-1/node-102"
					}
				}
			}
		]
	}
}
