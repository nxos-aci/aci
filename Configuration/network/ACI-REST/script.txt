#Configure a Network under specific tenant

#Note: Replace "<tenant-name>" with the tenant name in the POST URL

POST URL: http://SWITCH_IP/api/mo/uni/tn-<tenant-name>.json
Content-Type: application/json
Cache-Control: no-cache
POST BODY :
{
    "fvCtx": {
        "attributes": {
            "name": "test_net"
        }
    }
}
